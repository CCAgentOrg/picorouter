Routing Rules

routing:
  # Use local for short prompts
  - if: short_prompt
    use_local: true
  
  # Use local for code
  - if: contains_code
    use_local: true
    models: [codellama]
  
  # Use specific provider for reasoning
  - if: contains_reasoning
    providers: [groq]
  
  # Python-specific
  - if: language:python
    models: [codellama]

Storage

# JSONL (default)
storage:
  backend: jsonl
  log_file: logs/requests.jsonl

# SQLite
storage:
  backend: sqlite
  db_path: logs/picorouter.db

# Turso/LibSQL
storage:
  backend: turso
  turso_url: libsql://my-db.turso.io
  turso_token: your-auth-token
