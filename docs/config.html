⚙️ Configuration

Full Example

profiles:
  chat:
    local:
      provider: ollama
      endpoint: http://localhost:11434
      models: [llama3]
    
    cloud:
      providers:
        kilo:
          models: [minimax/m2.5:free]
        groq:
          models: [llama-3.1-70b-versatile]
        openai:
          models: [gpt-4o-mini]
        anthropic:
          models: [claude-3-haiku-20240307]
        google:
          models: [gemini-1.5-flash]
        mistral:
          models: [mistral-small-latest]
        together:
          models: [meta-llama/Llama-3-70b-chat-hf]
    
    routing:
      - if: short_prompt
        use_local: true
      - if: contains_code
        use_local: true
        models: [codellama]
    
    yolo: false

default_profile: chat

server:
  host: 0.0.0.0
  port: 8080
